class Solution {
    int ans = 0;
public:
    TreeNode* convertBST(TreeNode* root) {
        if(root){
            convertBST(root->right);
            int temp = root->val; 
            root->val += ans;
            ans += temp;
            convertBST(root->left);
        }
        return root;
    }
};
