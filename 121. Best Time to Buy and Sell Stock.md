class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int len = prices.size();
        if(len == 0) return 0;
        int Min = prices[0];
        int num = 0;
        for(int i = 1; i < len; i++){
            if(prices[i] < Min){//该元素比它前面的所有元素都要小，则这一天不能出售
                Min = prices[i];//修改最小值
            }else{
                num = max(num,prices[i] - Min);//要求最大的那个
            }
        }
        return num;
    }
};
