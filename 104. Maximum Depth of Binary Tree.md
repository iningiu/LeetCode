### 思路1  
求二叉树的高度。
```
class Solution { //递归写法
public:
    int maxDepth(TreeNode* root) {
        int H,HL,HR;
        if(!root)
            return 0;
        else{
            HL = maxDepth(root->left);
            HR = maxDepth(root->right);
            H = HL > HR ? HL : HR;
            return H + 1;
        }
    }
};
```
